# G(t) — The Genius Unit as executable function

def G(t, subject):
    APR  = subject.assembly_projection_rate   # novel synthesis events / session
    CI   = subject.coherence_index            # 0.0 → 1.0 signal integrity
    DCTV = subject.domain_cross_transfer_vel  # non-adjacent field jump speed

# QTAC₇ Stone Score
def stone_score(subject):
    weights = [1.618, 2.618, 4.236, 6.854, 11.090, 17.944, 29.034]  # φ^1 through φ^7
    operators = [
        subject.CMPD,   # Compounding Velocity
